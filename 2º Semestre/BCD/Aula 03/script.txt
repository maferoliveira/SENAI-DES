CREATE DATABASE loja;

USE loja;

CREATE TABLE Usuarios (
      id INTEGER AUTO_INCREMENT PRIMARY KEY,
      nome VARCHAR(160),
      nascimento DATE,
      endereco VARCHAR(100),
      telefone VARCHAR(11)
);

CREATE TABLE Pedidos (
      id INTEGER AUTO_INCREMENT PRIMARY KEY,
      clienteId INTEGER,
      datahora DATETIME,
      nomeProduto VARCHAR(50),
      valor DECIMAL(4,3),
      FOREIGN KEY (clienteId) REFERENCES Usuarios(id)
);
